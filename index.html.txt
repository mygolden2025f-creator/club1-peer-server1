  <h1>🌍 AI Language Voice Club</h1>
  <div id="chat">
    <div id="messages"></div>
    <button id="startBtn">🎤 Start Speaking</button>
    <button id="stopBtn">🛑 Stop</button>
  </div>

  <script>
    // استخدام Web Speech API
    const startBtn = document.getElementById("startBtn");
    const stopBtn = document.getElementById("stopBtn");
    const messages = document.getElementById("messages");

    let recognition;
    if ("webkitSpeechRecognition" in window) {
      recognition = new webkitSpeechRecognition();
      recognition.lang = "en-US"; // اللغة الأساسية (يمكن تغييرها)
      recognition.continuous = true;
      recognition.interimResults = true;

      recognition.onresult = function(event) {
        let transcript = "";
        for (let i = event.resultIndex; i < event.results.length; i++) {
          transcript += event.results[i][0].transcript;
        }
        addMessage("Me", transcript, "me");

        // الترجمة (باستخدام Google Translate API أو أي API لاحقاً)
        // حاليا فقط عرض النص
      };

      recognition.onerror = function(event) {
        console.error("Error:", event.error);
      };
    } else {
      alert("❌ المتصفح لا يدعم التعرف على الصوت.");
    }

    startBtn.onclick = () => { recognition.start(); };
    stopBtn.onclick = () => { recognition.stop(); };

    function addMessage(user, text, cls) {
      const div = document.createElement("div");
      div.classList.add("msg", cls);
      div.textContent = `${user}: ${text}`;
      messages.appendChild(div);
      messages.scrollTop = messages.scrollHeight;
    }
  </script>
